import { Tabs, TabItem } from '@astrojs/starlight/components';

Use the following citation formats when referencing ScarfBench:

<Tabs>
  <TabItem label="BibTeX">
```bibtex
@misc{scarfbench,
  title        = {ScarfBench},
  author       = {{IBM}},
  year         = {2026},
  howpublished = {\url{https://github.com/scarfbench}},
  note         = {GitHub repository}
}
```
  </TabItem>
  <TabItem label="APA">
```text
IBM. (2026). ScarfBench [Source code]. GitHub. https://github.com/scarfbench
```
  </TabItem>
  <TabItem label="MLA">
```text
IBM. ScarfBench. GitHub, 2026, https://github.com/scarfbench.
```
  </TabItem>
</Tabs>
